Model Description

This is a one compartment oral absorption model with Weight and Metabolizing capacity as covariates. isPM takes a value of either 0 which indicates is not a poor metabolizer or 1 which indicates is a poor metabolizer.

oral_one_cmpt = @model begin
  @param   begin
    tvKa     ∈ RealDomain(lower=0)
    tvCl     ∈ RealDomain(lower=0)
    tvVc     ∈ RealDomain(lower=0)
    pmoncl   ∈ RealDomain(lower= -0.99)
    Ω        ∈ PDiagDomain(3)
  end

  @random begin
    η        ~ MvNormal(Ω)
  end

  @covariates isPM wt

  @pre begin
    poor_met = isPM
    Weight   = wt
    Cl       = tvCl*(1 + pmoncl * poor_met) * (Weight/70)^0.75 * exp(η[1])
    Vc       = tvVc*(wt/70) * exp(η[2])
    Ka       = tvKa * exp(η[3])
  end

  @dynamics begin
    Depot'   = -Ka*Depot
    Central' =  Ka*Depot - (Cl/Vc)*Central
  end

  @derived begin
    cp       = @. 1000*(Central/Vc)
  end
end
